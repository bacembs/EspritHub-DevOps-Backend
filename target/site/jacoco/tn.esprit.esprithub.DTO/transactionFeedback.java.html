public class transactionFeedback {
    List<Transaction> transactions;
    int banned ;

    int days ;

        this.banned = banned;

        this.transactions = transactions;

        return transactions;

        this.transactions = transactions;

    public int getBanned() {

    }

    public void setBanned(int banned) {
        this.banned = banned;
    }

    public int getDays() {
        return days;
    }

    public void setDays(int days) {
        this.days = days;
    }
}
